#include <iostream>
using namespace std;
//passing array to an function


class ArrayExample
{
  public:


    void printArray(int input[],int size) //==> *input
    {
      for(int i=0;i<size;i++)
      {
          cout<<input[i]<<" ";
      }
       cout<<endl;
    }


};


int main()
{
    ArrayExample obj;


    int arr1[5]={50,60,700,80,90};
    int arr2[12]={2,40,6,8,10,85,45,65,7,14,12,16};
    int arr3[6]={12,78,654,145,123,789};
    //obj.printArray(arr1);


    cout<<"Elements in arr1 : ";
    obj.printArray(arr1,5);


    cout<<endl<<"Elements in arr2 : " ;
    obj.printArray(arr2,12);


    cout<<endl<<"Elements in arr3 : " ;
    obj.printArray(arr3,6);
    return 0;
}
