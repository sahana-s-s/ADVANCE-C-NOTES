#include<iostream>
using namespace std;


int main()
{
    char ch;
    cout<<"enter the character\n";
    cin>>ch;


    switch(ch) //switch(2)
    {
       case 'a':
       case 'e':
       case 'i':
       case 'o':
       case 'u': cout<<"vowel\n";
                 break;


      default: cout<<"Consonant!!!\n";
               break;
    }
    return 0;
}
