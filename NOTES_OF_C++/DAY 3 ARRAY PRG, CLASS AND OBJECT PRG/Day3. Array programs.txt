class ArrayExample
{
public:
    int arr[5];


    void accept()
    {
      cout<<"enter 5 elements for an array "<<endl;
      cin>>arr[0];
      cin>>arr[1];
      cin>>arr[2];
      cin>>arr[3];
      cin>>arr[4];
    }


    void disp()
    {
       cout<<"\nArray Elements are: "<<endl;


       for(int i : arr) //foreach
        //each element from an array is given to i
        {
          cout<< i <<endl;
        }
    }
};


int main()
{
  ArrayExample obj;
  obj.accept();
  obj.disp();
  return 0;
