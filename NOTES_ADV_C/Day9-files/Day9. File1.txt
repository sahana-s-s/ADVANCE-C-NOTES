#include <stdio.h>
//#define EOF // -1 present inside stdio.h


int main()
{
    FILE *fp;
    char ch;


    fp = fopen("D:/TEACHING/CRANES/FILES/mine.txt","r");


    if(fp==NULL)
    {
        printf("error opening file\n");
        return 1;
    }


    //read characters from file until condi is true


    while((ch = fgetc(fp)) != EOF)
    {
        putchar(ch);//output
        //printf("%c",ch);
    }


    fclose(fp);
    return 0;
}
//fgetc => EOF
